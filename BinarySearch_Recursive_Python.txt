def binarySearch(arr,l,r,x):
    if r>=l:
        mid = l + (r-l)//2
        if arr[mid]==x:
            return mid
        elif arr[mid]>x:
            return binarySearch(arr,l,mid-1,x)
        else:
            return binarySearch(arr,mid+1,r,x)
    else:
        return -1

arr = [10,20,30,40,50]
n = len(arr)
key = 80
print(binarySearch(arr,0,n,key))
